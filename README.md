### mongodb OK
```mongo
db.quotestoscrape.aggregate([
        {$group: {_id: {text: '$text', author: '$author'}, count: {$sum: 1}, dups: {$addToSet: '$_id'}}},
        {$match: {count: {$gt: 1}}}
]).forEach(function(doc){
    doc.dups.shift();
    db.quotestoscrape.remove({_id: {$in: doc.dups}});
})
```
